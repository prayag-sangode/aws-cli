# Push the commit with the empty Git directory to remote repositories such as GitHub or GitLab
~$ mkdir empty-directory

~$ cd empty-directory

~$ touch .gitkeep

~$ git add .

~$ git commit -m "Commit empty folder in Git with gitkeep"

~$ git push origin
